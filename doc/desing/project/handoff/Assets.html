Integración Flutter

// pubspec.yaml — declara la carpeta
flutter:
  assets:
    - assets/species/
    - assets/icon/

// widget — usa flutter_svg
import 'package:flutter_svg/flutter_svg.dart';

class PetAvatar extends StatelessWidget {
  final String species; // 'dog' | 'cat' | 'rabbit' | 'bird'
  final Color tint;
  const PetAvatar({required this.species, this.tint = AppColors.primary});

  @override
  Widget build(BuildContext context) {
    return Container(
      width: 56, height: 56,
      decoration: BoxDecoration(
        shape: BoxShape.circle,
        color: AppColors.primarySoft,
      ),
      child: Center(
        child: SvgPicture.asset(
          'assets/species/$species.svg',
          width: 36, height: 36,
          colorFilter: ColorFilter.mode(tint, BlendMode.srcIn),
        ),
      ),
    );
  }
}
Para iOS app icon: Xcode requiere PNG en múltiples resoluciones. Usa un exportador (ej. appicon.co) o Figma para generar el set completo a partir de app-icon.svg. Tamaños necesarios: 20pt, 29pt, 40pt, 60pt, 76pt, 83.5pt, 1024pt — todos en @1x @2x @3x.
